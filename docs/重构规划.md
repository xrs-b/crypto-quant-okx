## 1. 项目目标

将现有单文件脚本重构为现代化、模块化、可扩展的量化交易系统。

## 4. 数据库设计 (SQLite)

```sql
-- 交易记录表
CREATE TABLE trades (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    symbol TEXT NOT NULL,
    side TEXT NOT NULL,           -- long / short
    entry_price REAL NOT NULL,
    exit_price REAL,
    quantity REAL NOT NULL,
    leverage INTEGER NOT NULL,
    pnl REAL,                     -- 盈亏
    pnl_percent REAL,              -- 盈亏比例
    status TEXT NOT NULL,          -- open / closed
    open_time DATETIME NOT NULL,
    close_time DATETIME,
    reason TEXT,                   -- 止损/止盈/手动
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 信号记录表
CREATE TABLE signals (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    symbol TEXT NOT NULL,
    signal_type TEXT NOT NULL,     -- buy / sell
    price REAL NOT NULL,
    strength INTEGER NOT NULL,      -- 0-100
    reasons TEXT NOT NULL,          -- JSON格式，记录各策略分析
    strategies_triggered TEXT,     -- 触发既策略列表
    filtered BOOLEAN DEFAULT FALSE,
    filter_reason TEXT,            -- 过滤原因
    executed BOOLEAN DEFAULT FALSE,
    executed_at DATETIME,
    trade_id INTEGER,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (trade_id) REFERENCES trades(id)
);

-- 策略分析记录表
CREATE TABLE strategy_analysis (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    signal_id INTEGER NOT NULL,
    strategy_name TEXT NOT NULL,
    strategy_type TEXT NOT NULL,  -- rsi / macd / ma / bollinger / ml
    triggered BOOLEAN NOT NULL,
    details TEXT,                  -- JSON格式，策略详细分析
    confidence REAL,               -- 置信度
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (signal_id) REFERENCES signals(id)
);

-- 持仓表
CREATE TABLE positions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    symbol TEXT NOT NULL UNIQUE,
    side TEXT NOT NULL,
    entry_price REAL NOT NULL,
    quantity REAL NOT NULL,
    leverage INTEGER NOT NULL,
    current_price REAL,
    unrealized_pnl REAL,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 系统配置表
CREATE TABLE settings (
    key TEXT PRIMARY KEY,
    value TEXT NOT NULL,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_trades_symbol ON trades(symbol);
CREATE INDEX idx_trades_status ON trades(status);
CREATE INDEX idx_trades_open_time ON trades(open_time);
CREATE INDEX idx_signals_symbol ON signals(symbol);
CREATE INDEX idx_signals_created ON signals(created_at);
CREATE INDEX idx_signals_executed ON signals(executed);
```
